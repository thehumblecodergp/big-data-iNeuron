-- Command to create a Database

CREATE DATABASE test;

-- number of DATABASES
SHOW DATABASES;

-- Go inside the DATABASE
USE test;

-- Create a TABLE
CREATE TABLE IF NOT EXISTS employee
(
    id INT,
    name VARCHAR(50)
);

-- list of all TABLES
SHOW TABLES;

-- To get the "CREATE TABLE" command of a different TABLE
SHOW CREATE TABLE employee;
